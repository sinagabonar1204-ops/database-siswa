let step = 1;
let dataSiswa = {};

  document.querySelectorAll('#progress div').forEach((el, i) => {
    el.className = "w-8 h-2 rounded-full " + (i+1===step ? "bg-emerald-400" : "bg-gray-600");
  });

  const container = document.getElementById('form-container');
  container.innerHTML = "";

  if(step===1){
    container.innerHTML = `
      <div class="text-center space-y-4">
        <img src="https://via.placeholder.com/100" class="mx-auto rounded-full"/>
        <h1 class="text-2xl font-bold">Selamat Datang Siswa Baru!</h1>
        <p>Silakan lengkapi profil Anda untuk database induk sekolah.</p>
        <button onclick="step++ ; renderStep()" class="bg-emerald-500 hover:bg-emerald-600 px-6 py-2 rounded-xl">Lengkapi Data Siswa</button>
      </div>`;
  }

  else if(step===2){
    container.innerHTML = `
      <h2 class="text-xl font-bold mb-4">Data Siswa</h2>
      <form id="formSiswa" class="space-y-2">
        <input placeholder="NISN" class="w-full p-2 rounded-xl text-black"/>
        <input placeholder="Nama Lengkap" class="w-full p-2 rounded-xl text-black"/>
        <input placeholder="Jenis Kelamin" class="w-full p-2 rounded-xl text-black"/>
        <input placeholder="Tempat Lahir" class="w-full p-2 rounded-xl text-black"/>
        <input type="date" placeholder="Tanggal Lahir" class="w-full p-2 rounded-xl text-black"/>
        <input placeholder="NIK" class="w-full p-2 rounded-xl text-black"/>
        <input placeholder="Sekolah Asal" class="w-full p-2 rounded-xl text-black"/>
        <input placeholder="NPSN" class="w-full p-2 rounded-xl text-black"/>
        <textarea placeholder="Alamat Lengkap" class="w-full p-2 rounded-xl text-black"></textarea>
        <input placeholder="Kecamatan" class="w-full p-2 rounded-xl text-black"/>
        <input placeholder="Kabupaten/Kota" class="w-full p-2 rounded-xl text-black"/>
        <input placeholder="Provinsi" class="w-full p-2 rounded-xl text-black"/>
        <input placeholder="No. HP Siswa" class="w-full p-2 rounded-xl text-black"/>
        <input placeholder="Status Yatim/Piatu" class="w-full p-2 rounded-xl text-black"/>
        <button type="button" onclick="simpanDataSiswa()" class="bg-emerald-500 px-6 py-2 rounded-xl">Selanjutnya: Data Orang Tua</button>
      </form>`;
  }

  else if(step===3){
    container.innerHTML = `
      <h2 class="text-xl font-bold mb-4">Data Orang Tua & Wali</h2>
      <p>Form ayah, ibu, wali (disederhanakan untuk contoh)</p>
      <button onclick="step++ ; renderStep()" class="bg-emerald-500 px-6 py-2 rounded-xl">Selanjutnya: Unggah Berkas</button>`;

function simpanDataSiswa() {
  dataSiswa = {
    nisn: document.querySelector('input[placeholder="NISN"]').value,
    nama: document.querySelector('input[placeholder="Nama Lengkap"]').value,
    jk: document.querySelector('input[placeholder="Jenis Kelamin"]').value,
    tempat_lahir: document.querySelector('input[placeholder="Tempat Lahir"]').value,
    tanggal_lahir: document.querySelector('input[type="date"]').value,
    nik: document.querySelector('input[placeholder="NIK"]').value,
    sekolah_asal: document.querySelector('input[placeholder="Sekolah Asal"]').value,
    npsn: document.querySelector('input[placeholder="NPSN"]').value,
    alamat: document.querySelector('textarea[placeholder="Alamat Lengkap"]').value,
    kecamatan: document.querySelector('input[placeholder="Kecamatan"]').value,
    kota: document.querySelector('input[placeholder="Kabupaten/Kota"]').value,
    provinsi: document.querySelector('input[placeholder="Provinsi"]').value,
    hp: document.querySelector('input[placeholder="No. HP Siswa"]').value,
    status_yatim: document.querySelector('input[placeholder="Status Yatim/Piatu"]').value
  };

  step++;
  renderStep();
}
async function submitData() {
  const payload = dataSiswa;

  try {
    const res = await fetch("https://script.google.com/macros/s/AKfycbzGiMLtZs2WxyOax2NWQ6O-derbWL5W8-UFKN11OtaxmkxTsnNA6jYrwxeXj2sGv1vn/exec", { //
      method: "POST",
      body: JSON.stringify(payload),
      headers: { "Content-Type": "text/plain;charset=utf-8" }
    });
    const text = await res.text();
    if(text==="SUCCESS"){
      step = 6;
      renderStep();
    } else {
      alert("Terjadi kesalahan: " + text);
    }
  } catch(err){
    alert("Gagal mengirim: " + err);
  }
}
